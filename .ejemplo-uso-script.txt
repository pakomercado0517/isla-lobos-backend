📝 PRÓXIMOS PASOS:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Inicia sesión en la aplicación
2. Cambia tu contraseña por una segura
3. Crea invitaciones para prestadores: POST /api/invitaciones
4. Los prestadores se registran con el código de invitación
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━


# ────────────────────────────────────────────────────────────────────────
# PASO 3: Login con las credenciales
# ────────────────────────────────────────────────────────────────────────

curl -X POST http://localhost:3000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "admin@conanp.gob.mx",
    "password": "ContraseñaSegura123!"
  }'

# Respuesta:
{
  "status": "success",
  "message": "Login exitoso",
  "data": {
    "user": {
      "id": "uuid-aqui",
      "nombre": "Administrador CONANP",
      "email": "admin@conanp.gob.mx",
      "rol": "conanp"
    },
    "token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
  }
}


# ────────────────────────────────────────────────────────────────────────
# PASO 4: Cambiar contraseña (RECOMENDADO)
# ────────────────────────────────────────────────────────────────────────

curl -X PUT http://localhost:3000/api/auth/change-password \
  -H "Authorization: Bearer <token-del-login>" \
  -H "Content-Type: application/json" \
  -d '{
    "password": "ContraseñaSegura123!",
    "newPassword": "NuevaContraseñaSegura456!"
  }'


# ────────────────────────────────────────────────────────────────────────
# PASO 5: Crear invitaciones para prestadores
# ────────────────────────────────────────────────────────────────────────

curl -X POST http://localhost:3000/api/invitaciones \
  -H "Authorization: Bearer <token>" \
  -H "Content-Type: application/json" \
  -d '{
    "email": "prestador@ejemplo.com",
    "rol": "prestador"
  }'

# Respuesta:
{
  "status": "success",
  "message": "Invitación creada exitosamente",
  "data": {
    "invitacion": {
      "codigo": "ABC123XYZ",
      "email": "prestador@ejemplo.com",
      "rol": "prestador",
      "expira_en": "2025-11-13"
    }
  }
}


# ────────────────────────────────────────────────────────────────────────
# PASO 6: Prestador se registra con el código
# ────────────────────────────────────────────────────────────────────────

curl -X POST http://localhost:3000/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "nombre": "Juan Pérez",
    "email": "prestador@ejemplo.com",
    "password": "Prestador123!",
    "telefono": "2291234567",
    "codigo_invitacion": "ABC123XYZ"
  }'


# ────────────────────────────────────────────────────────────────────────
# ¡LISTO! Sistema configurado y usuarios creados
